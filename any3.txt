// 1. SELECTION SORT


void runSelectionSort() {
   cout << "\n--- Running Selection Sort ---\n";
   int arr[] = {64, 25, 12, 22, 11};
   int n = 5;


   cout << "Original: ";
   for (int i = 0; i < n; i++) cout << arr[i] << " ";
   cout << endl;


   for (int i = 0; i < n - 1; i++) {
       int min_idx = i;
       for (int j = i + 1; j < n; j++) {
           if (arr[j] < arr[min_idx])
               min_idx = j;
       }
       swap(arr[min_idx], arr[i]);
   }


   cout << "Sorted:   ";
   for (int i = 0; i < n; i++) cout << arr[i] << " ";
   cout << endl;
}




// 2. JOB SCHEDULING (The Freelancer)


struct Job {
   int id;     
   int deadline;
   int profit;
};


bool compareJobs(Job a, Job b) {
   return (a.profit > b.profit);
}


void runJobScheduling() {
   cout << "\n--- Running Job Scheduling ---\n";
   Job arr[] = { {1, 4, 20}, {2, 1, 10}, {3, 1, 40}, {4, 1, 30} };
   int n = 4;




   sort(arr, arr + n, compareJobs);


   int result[n];
   bool slot[n]; 


   for (int i = 0; i < n; i++) slot[i] = false;


   for (int i = 0; i < n; i++) {
       for (int j = min(n, arr[i].deadline) - 1; j >= 0; j--) {
           if (slot[j] == false) {
               result[j] = i; 
               slot[j] = true;
               break;
           }
       }
   }


   cout << "Jobs selected for Max Profit: ";
   for (int i = 0; i < n; i++) {
       if (slot[i]) cout << "Job" << arr[result[i]].id << " ";
   }
   cout << endl;
}


// 3. PRIM'S ALGORITHM (The Cable Guy)
#define V 5


void runPrimMST() {
   cout << "\n--- Running Prim's MST ---\n";
   int graph[V][V] = {
       { 0, 2, 0, 6, 0 },
       { 2, 0, 3, 8, 5 },
       { 0, 3, 0, 0, 7 },
       { 6, 8, 0, 0, 9 },
       { 0, 5, 7, 9, 0 }
   };


   int parent[V];
   int key[V];    
   bool mstSet[V];


   for (int i = 0; i < V; i++) { key[i] = 999; mstSet[i] = false; }


   key[0] = 0;    
   parent[0] = -1;


   for (int count = 0; count < V - 1; count++) {
       int min = 999, u;
       for (int v = 0; v < V; v++)
           if (mstSet[v] == false && key[v] < min)
               min = key[v], u = v;


       mstSet[u] = true;


       for (int v = 0; v < V; v++)
           if (graph[u][v] && mstSet[v] == false && graph[u][v] < key[v])
               parent[v] = u, key[v] = graph[u][v];
   }


   cout << "Edge \tWeight\n";
   for (int i = 1; i < V; i++)
       cout << parent[i] << " - " << i << " \t" << graph[i][parent[i]] << " \n";
}


// 4. DIJKSTRA'S ALGORITHM (The GPS)
void runDijkstra() {
   cout << "\n--- Running Dijkstra's Shortest Path ---\n";
   int graph[V][V] = {
       { 0, 10, 0, 0, 5 },
       { 10, 0, 1, 0, 2 },
       { 0, 1, 0, 4, 0 },
       { 0, 0, 4, 0, 2 },
       { 5, 2, 0, 2, 0 }
   };


   int src = 0;


   int dist[V];   
   bool Set[V];


   for (int i = 0; i < V; i++) { dist[i] = 999; Set[i] = false; }


   dist[src] = 0;


   for (int count = 0; count < V - 1; count++) {
       int min = 999, u;
       for (int v = 0; v < V; v++)
           if (Set[v] == false && dist[v] <= min)
               min = dist[v], u = v;
              


       Set[u] = true;


       for (int v = 0; v < V; v++)
           if (!Set[v] && graph[u][v] && dist[u] != 999
               && dist[u] + graph[u][v] < dist[v]) {
               dist[v] = dist[u] + graph[u][v];
           }
   }


   cout << "City \tDistance from Source (0)\n";
   for (int i = 0; i < V; i++)
       cout << i << " \t" << dist[i] << endl;
}
