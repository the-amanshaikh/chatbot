#include <iostream>
#include <vector>
#include <queue>


using namespace std;


vector<int> cityConnections[100];
bool visitedDFS[100];
bool visitedBFS[100];




void DFS(int city, int parent) {
   visitedDFS[city] = true;


   if (parent == -1)
       cout << "City " << city << " is infected first." << endl;
   else
       cout << "City " << city << " is bitten by zombie from City " << parent << "." << endl;


   for (int neighbor : cityConnections[city]) {
       if (!visitedDFS[neighbor]) {
           DFS(neighbor, city);
       }
   }
}




void BFS(int startCity) {
   queue<int> q;
   q.push(startCity);
   visitedBFS[startCity] = true;


   int wave = 0;


   cout << "\nBFS Spread Order (Wave by Wave):" << endl;


   while (!q.empty()) {
       int size = q.size();
       cout << "Wave " << wave << ": ";


       while (size--) {
           int current = q.front();
           q.pop();
           cout << "City " << current << " ";


           for (int neighbor : cityConnections[current]) {
               if (!visitedBFS[neighbor]) {
                   visitedBFS[neighbor] = true;
                   q.push(neighbor);
               }
           }
       }


       cout << endl;
       wave++;
   }
}


int main() {
   int cities, roads;
   cin >> cities >> roads;


   for (int i = 0; i < roads; i++) {
       int u, v;
       cin >> u >> v;
       cityConnections[u].push_back(v);
       cityConnections[v].push_back(u);
   }


   cout << "\nDFS Infection Chain:\n";
   DFS(0, -1);


   cout << endl;
   BFS(0);


   return 0;
}
